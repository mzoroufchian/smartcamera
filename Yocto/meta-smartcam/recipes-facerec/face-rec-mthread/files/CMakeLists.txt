cmake_minimum_required(VERSION 2.8)
project( Mthr_face_rec )
find_package( OpenCV REQUIRED )
find_package (Threads REQUIRED)

file( GLOB DRIV_SOURCES driv/*.c )
file( GLOB DRIV_HEADERS driv/*.h )
add_executable( Mthr_face_rec Mthr_face_rec.cpp ${DRIV_SOURCES} ${DRIV_HEADERS})
target_link_libraries( Mthr_face_rec ${OpenCV_LIBS} ${CMAKE_THREAD_LIBS_INIT})
